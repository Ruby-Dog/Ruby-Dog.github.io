指標-交換變數的值
2021/06/09

交換值小練習

#include <stdio.h>

void main(){
    
    int x,y,temp;
    int *p1,*p2;
    printf("請輸入x的值:");
    scanf("%d",&x);   
    /*scanf("格式控制字符串", 地址表列);
    把輸入的東西掃描後,存入該地址裡*/
    
    printf("請輸入y的值:");
    scanf("%d",&y);
    
    printf("交換前,x = %d  y = %d \n",x,y);
    p1 = &x;
    p2 = &y;
    
    if(*p1<*p2){
        temp = *p1;
        *p1=*p2;
        *p2=temp;
    }
    
    printf("交換後,x = %d  y = %d \n",x,y);
}
結果:
請輸入x的值:10
請輸入y的值:15
交換前,x = 10 y = 15
交換後,x = 15 y = 10
